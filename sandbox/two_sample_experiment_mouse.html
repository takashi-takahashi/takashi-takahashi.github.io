@numba.jit
def do_bootstrap_experiment(treatment, control, B):
    first_moment = 0.0
    second_moment= 0.0
    diff_list = []
    for _ in range(B):
        z = np.random.choice(treatment, 7)
        y = np.random.choice(control, 9)
        diff = np.mean(z) - np.mean(y)
        first_moment += diff
        second_moment += diff ** 2
        diff_list.append(diff)
    return np.sqrt((second_moment - first_moment ** 2 / B) / (B-1)), diff_list
